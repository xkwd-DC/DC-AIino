# Python 3.10 / Ubuntu 22.04 / 2vCPU 4GB (CPU-only inference)
# 注意：tensorflow 用 tensorflow-cpu，体积 ~300MB 而非 ~600MB+，避免拖入 CUDA 依赖
# 4GB 内存对 TF 推理足够；如后续上 GPU 部署再换回 tensorflow

# --- Web 框架 ---
flask~=3.0.3
flask-cors~=5.0.0
gunicorn~=23.0.0

# --- 配置 ---
python-dotenv~=1.0.1

# --- 数据处理 ---
# numpy 暂封顶 2.1：SHAP 0.46 与 xgboost 2.1 在 numpy 2.x 上偶发兼容问题
numpy>=1.26,<2.1
pandas~=2.2.2

# --- 机器学习 ---
scikit-learn~=1.5.1
xgboost~=2.1.1
# Issue #39: 训练 artifact 由 keras 3.14 保存,Dense 层 config 带 quantization_config。
# tensorflow-cpu 2.17 (keras 3.5) 加载会报 `Unrecognized keyword arguments passed
# to Dense: {'quantization_config': None}`。bump 到 2.18+ 取得更新的 keras。
# 同时 api/predict.py 的 _apply_keras_compat_shim() 兜底 — python 3.10 上
# pip 装的 keras 上限仍是 3.12,补丁 strip 掉这个 unknown kwarg 以兼容加载。
tensorflow-cpu>=2.18,<2.20
shap~=0.46.0

# --- API 限速 (生产必备, Issue #26 HIGH#3) ---
# in-memory 后端默认;多 worker 部署需配 RATELIMIT_STORAGE_URI=redis://...
flask-limiter~=3.8.0

# --- Parquet 引擎 (训练 / 推理读 paper_panel_v3.parquet 必需) ---
# pandas 默认尝试 pyarrow → fastparquet,二者皆缺则 read_parquet 抛 ImportError。
# 训练时 train_*.py 直接 pd.read_parquet,生产部署里 services.inference 不读 parquet,
# 但启动 sanity / smoke check 仍可能需要,所以入 requirements。
pyarrow~=17.0
